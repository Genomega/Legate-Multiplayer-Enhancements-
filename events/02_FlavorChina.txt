## Imperial Capital
country_event = {
	id = chinese_flavor.5
	title = "chinese_flavor.5.t"
	desc = "chinese_flavor.5.d"
	picture = DEATH_OF_HEIR_eventPicture
	
	is_triggered_only = yes
	
	option = { # Beijing
		name = "chinese_flavor.5.a"
		trigger = {
			owns = 1816	
		}		
		1816 = {
			move_capital_effect = yes
			add_center_of_trade_level = 1
			add_base_manpower = 1
			add_base_tax = 1
			add_base_production = 1
		}	
	}
	
	option = { # Nanjing
		name = "chinese_flavor.5.b"
		trigger = {
			owns = 1821	
		}		
		1821 = {
			move_capital_effect = yes
			add_center_of_trade_level = 1
			add_base_manpower = 1
			add_base_tax = 1
			add_base_production = 1
		}	
	}
	
	option = { # Luoyang
		name = "chinese_flavor.5.c"
		trigger = {
			owns = 1836	
		}		
		1836 = {
			move_capital_effect = yes
			center_of_trade = 2
			add_base_manpower = 3
			add_base_tax = 3
			add_base_production = 3
		}	
	}

	option = { # Jingzhou
		name = "chinese_flavor.5.dd"
		trigger = {
			owns = 2172	
		}		
		2172 = {
			add_province_modifier = {
				name = "chinese_great_capital"
				duration = 18250
			}			
			move_capital_effect = yes
			add_base_manpower = 2
			add_base_tax = 2
			add_base_production = 2
		}	
	}	
	
	option = { # Canton
		name = "chinese_flavor.5.e"
		trigger = {
			owns = 667	
		}		
		667 = {
			move_capital_effect = yes
			add_base_manpower = 1
			add_base_tax = 1
			add_base_production = 1
		}	
	}

	option = { # Current Capital 
		name = "chinese_flavor.5.f"
		trigger = {
			NOT = {
				667 = {
					 is_capital_of = ROOT
				}
			}
			NOT = {
				2172 = {
					 is_capital_of = ROOT
				}
			}
			NOT = {
				1836 = {
					 is_capital_of = ROOT
				}
			}
			NOT = {
				1821 = {
					 is_capital_of = ROOT
				}
			}
			NOT = {
				1816 = {
					 is_capital_of = ROOT
				}
			}			
		}		
		capital_scope = {
			add_province_modifier = {
				name = "chinese_great_capital"
				duration = 18250
			}	
			add_center_of_trade_level = 1
			center_of_trade = 1
			move_capital_effect = yes
			add_base_manpower = 3
			add_base_tax = 3
			add_base_production = 3
		}	
	}		
}
